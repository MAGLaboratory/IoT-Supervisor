89pb_t PetitPortRegWrite(pu8_t Address, pu16_t Data)
90{
91 // check if you can write to this
92 if (Address > eMMW_HR_CFG && cfgSmS != eCFG_Cache)
93 {
94 return 0;
95 }
96 // the status register only accepts two values, and petting the watchdog
97 // automatically enables it
98 if (Address == eMMW_HR_STA)
99 {
100 if (Data == C_WDT_PET)
101 {
102 mbWDTpet = true;
103 mbWDTen = true;
104 }
105 else if (Data == C_WDT_DIS)
106 {
107 mbWDTen = false;
108 }
109 else
110 {
111 return 0;
112 }
113 }
114 // you can enter whatever password you want
115 if (Address == eMMW_HR_CFG)
116 {
117 pw = Data;
118 pw_flag = true;
119 }
120 // it's perfectly valid to enter 0 for this since none of the bytewise
121 // changes will be applied
122 if (Address == eMMW_HR_MB)
123 {
124 if ((Data & 0xFF) > 0)
125 {
126 // but if you enter an SID that is too large, you will get an error
127 if ((Data & 0xFF) < 248)
128 {
129 cfg.sid = Data & 0xFF;
130 }
131 else
132 {
133 return 0;
134 }
135 }
136 if (Data >> 8 > 0)
137 {
138 if (Data >> 8 < eMMW_B_NUM)
139 {
140 cfg.baud = Data >> 8;
141 }
142 }
143 }
144 // the WDT timeout must be at least one minute.
145 if (Address == eMMW_HR_WDT)
146 {
147 if (Data > 0)
148 {
149 cfg.wdto = Data;
150 }
151 else
152 {
153 return 0;
154 }
155 }
156 // the new password must be valid
157 if (Address == eMMW_HR_PW)
158 {
159 if (Data != 0 && Data != C_CMD_COMMIT && Data != C_CMD_CANCEL)
160 {
161 cfg.pw = Data;
162 }
163 else
164 {
165 return 0;
166 }
167 }
168
169 return 1;
170}

178pb_t PetitPortRegRead(pu8_t Address, pu16_t* Data)
179{
180 // check if you can access this
181 if (Address > eMMW_HR_CFG &&
182 (cfgSmS != eCFG_Cache && cfgSmS != eCFG_Commit))
183 {
184 return 0;
185 }
186 if (Address == eMMW_HR_STA)
187 {
188 *Data = sv_dev_sta.b;
189 }
190 if (Address == eMMW_HR_CFG)
191 {
192 *Data = cfgSmS;
193 }
194 if (Address == eMMW_HR_MB)
195 {
196 *Data = cfg.baud << 8 | cfg.sid;
197 }
198 if (Address == eMMW_HR_WDT)
199 {
200 *Data = MB_WD_TIMEOUT;
201 }
202 if (Address == eMMW_HR_PW)
203 {
204 *Data = cfg.pw;
205 }
206 return 1;
207}
