107{

109 static uint8_t sCounter = 0;
110
111 // determine input state
112 bool VinCmp = CMP0CN0 & CMP0CN0_CPOUT__BMASK;
113
114 // default output states
115 nReset = false;
116 modbusWdtSmEn = false;
117
118 // transitions
119 switch (sv_dev_sta.v.vinSmS)
120 {
121 case eVIN_Init:
122 if (cprif == true || VinCmp == true)
123 {
124 sv_dev_sta.v.vinSmS = eVIN_VLow;
125 }
126 else if (sCounter >= SCOUNTER_ONE_SECOND * 2 - 1)
127 {
128 sv_dev_sta.v.vinSmS = eVIN_OK;
129 }
130 break;
131 case eVIN_VLow:
132 if (VinCmp == false)
133 {
134 sv_dev_sta.v.vinSmS = eVIN_Init;
135 }
136 break;
137 case eVIN_OK:
138 if (cprif == true || VinCmp == true)
139 {
140 sv_dev_sta.v.vinSmS = eVIN_VLow;
141 sv_dev_sta.v.lastRstS = eLR_VSM;
142 }
143 else if (modbusWdtExp == true)
144 {
145 sv_dev_sta.v.vinSmS = eVIN_VLow;
146 sv_dev_sta.v.lastRstS = eLR_WDT;
147 }
148 break;
149 default:
150 sv_dev_sta.v.vinSmS = eVIN_VLow;
151 break;
152 }
153
154 // outputs
155 switch (sv_dev_sta.v.vinSmS)
156 {
157 case eVIN_Init:
158 sCounter++;
159 modbusWdtExp = false;
160 break;
161 case eVIN_VLow:
162 sCounter = 0;
163 break;
164 case eVIN_OK:
165 nReset = true;
166 modbusWdtSmEn = true;
167 sCounter = 0;
168 break;
169 }
170
171 // output signals
172 RESET_P = nReset;
173 nLED = nReset;
174
175 // reset flags
176 cprif = false;

355bool ccfg_check(void)
356{
357 uint8_t i = 0;
358 calc_cfg_crc = 0xFFFF;
359 // check header
360 for (i = 0; i < C_CFG_HEADER_LEN; i++)
361 {
362 if (*(pFLASH_CONF + i) != ex_cfg_header[i])
363 {
364 sv_dev_sta.v.verifSt = eVS_Setup;
365 return false;
366 }
367 KirisakiCRC16Calc(ex_cfg_header[i], &calc_cfg_crc);
368 }
369
370 // check contents
371 // modbus SID
372 i = *(pFLASH_CONF + C_CFG_HEADER_LEN);
373 if(i < C_SID_MIN || i > C_SID_MAX)
374 {
375 sv_dev_sta.v.verifSt = eVS_Cfg;
376 return false;
377 }
378 else
379 {
380 cfg.sid = i;
381 KirisakiCRC16Calc(i, &calc_cfg_crc);
382 }
383 // modbus baud config
384 i = *(pFLASH_CONF + C_CFG_HEADER_LEN + 1);
385 if (i == 0 || i >= eMMW_B_NUM)
386 {
387 sv_dev_sta.v.verifSt = eVS_Cfg;
388 return false;
389 }
390 else
391 {
392 cfg.baud = i;
393 KirisakiCRC16Calc(i, &calc_cfg_crc);
394 }
395 // modbus WD timeout
396 i = *(pFLASH_CONF + C_CFG_HEADER_LEN + 2);
397 if (i == 0)
398 {
399 sv_dev_sta.v.verifSt = eVS_Cfg;
400 return false;
401 }
402 else
403 {
404 cfg.wdto = i;
405 KirisakiCRC16Calc(i, &calc_cfg_crc);
406 }
407 // password
408 i = *(pFLASH_CONF + C_CFG_HEADER_LEN + 3);
409 cfg.pw = i;
410 KirisakiCRC16Calc(i, &calc_cfg_crc);
411 i = *(pFLASH_CONF + C_CFG_HEADER_LEN + 4);
412 cfg.pw |= (uint16_t)i << 8;
413 KirisakiCRC16Calc(i, &calc_cfg_crc);
414 if (cfg.pw == C_CMD_COMMIT || cfg.pw == C_CMD_CANCEL)
415 {
416 sv_dev_sta.v.verifSt = eVS_Cfg;
417 return false;
418 }
419 // check CFG CRC
420 i = *(pFLASH_CONF + C_CFG_DATA_END);
421 if ((calc_cfg_crc & 0xFF) != i)
422 {
423 sv_dev_sta.v.verifSt = eVS_Cfg;
424 return false;
425 }
426 else
427 {
428 i = *(pFLASH_CONF + C_CFG_DATA_END + 1);
429 if (calc_cfg_crc >> 8 != i)
430 {
431 sv_dev_sta.v.verifSt = eVS_Cfg;
432 return false;
433 }
434 }
435 return true;
436}

483void cfg_load()
484{
485 uint8_t code* addr;
486 uint8_t i = 0;
487 bool cfg_ok = true;
488
489 cfg_ok = ccfg_check();
490
491 // calculate PROG CRC
492 calc_prog_crc = 0xFFFF;
493 for (addr = 0; addr < C_FOUND_PROG_END; addr++)
494 {
495 KirisakiCRC16Calc(*addr, &calc_prog_crc);
496 }
497
498 // check PROG CRC
499 if (sv_dev_sta.v.verifSt != eVS_Setup)
500 {
501 i = *(pFLASH_CONF + C_CFG_C_CRC_END);
502 if ((calc_prog_crc & 0xFF) != i)
503 {
504 sv_dev_sta.v.verifSt = eVS_Prog;
505 }
506 else
507 {
508 i = *(pFLASH_CONF + C_CFG_C_CRC_END + 1);
509 if (calc_prog_crc >> 8 != i)
510 {
511 sv_dev_sta.v.verifSt = eVS_Prog;
512 }
513 }
514 }
515
516 if (cfg_ok == false)
517 {
518 // load the default configuration structure
519 cfg.sid = default_cfg.sid;
520 cfg.baud = default_cfg.baud;
521 cfg.wdto = default_cfg.wdto;
522 cfg.pw = default_cfg.pw;
523 }
524

526 mmw_init(cfg.sid, cfg.baud);
527 MB_WD_TIMEOUT = cfg.wdto;
528
529 cfgSmS = eCFG_Idle;

541 {

552 // wdt top
553 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, cfg.wdto);
554 KirisakiCRC16Calc(cfg.wdto, &calc_cfg_crc);
555 // pw
556 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, cfg.pw & 0xFF);
557 KirisakiCRC16Calc(cfg.pw & 0xFF, &calc_cfg_crc);
558 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, cfg.pw >> 8);
559 KirisakiCRC16Calc(cfg.pw >> 8, &calc_cfg_crc);
560 // cfg crc
561 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, calc_cfg_crc & 0xFF);
562 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, calc_cfg_crc >> 8);
563 // prog crc
564 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, calc_prog_crc & 0xFF);
565 FLASH_ByteWrite(C_FLASH_CONF + cfg_write_idx++, calc_prog_crc >> 8);
566
567 // reset to complete
568 RSTSRC |= RSTSRC_SWRSF__SET;
569 // uh, set the state machine to idle in case we have not reset
570 cfgSmS = eCFG_Idle;

585void CFGsm()
586{
587 // transitions
588 switch (cfgSmS)
589 {
590 case eCFG_Load:
591 // transitions handled in-function
592 break;
593 case eCFG_Idle:
594 if (pw_flag)
595 {
596 if (pw == cfg.pw)
597 {
598 cfgSmS = eCFG_Cache;
599 }
600 pw_flag = false;
601 }
602 break;
603 case eCFG_Cache:
604 if (pw_flag && Petit_RxTx_State == PETIT_RXTX_RX && dir_tx == false)
605 {
606 if (pw == C_CMD_COMMIT)
607 {
608 if (dcfg_check())
609 {
610 cfgSmS = eCFG_Commit;
611 mmw_init(cfg.sid, cfg.baud);
612 MB_WD_TIMEOUT = cfg.wdto;
613 }
614 }
615 if (pw == C_CMD_CANCEL)
616 {
617 cfgSmS = eCFG_Idle;
618 cfg = default_cfg;
619 }
620 pw_flag = false;
621 }
622 break;
623 case eCFG_Commit:
624 if (pw_flag && Petit_RxTx_State == PETIT_RXTX_RX && dir_tx == false)
625 {
626 if (pw == cfg.pw)
627 {
628 cfgSmS = eCFG_Erase;
629 }
630 if (pw == C_CMD_CANCEL)
631 {
632 cfgSmS = eCFG_Idle;
633 }
634 pw_flag = false;
635 }
636 break;
637 case eCFG_Erase:
638 cfgSmS = eCFG_Write;
639 break;
640 case eCFG_Write:
641 // transitions handled in function
642 break;
643 default:
644 cfgSmS = eCFG_Idle;
645 break;
646 }
647
648 // outputs
649 switch (cfgSmS)
650 {
651 case eCFG_Load:
652 cfg_load();
653 run_petitmodbus = false;
654 break;
655 case eCFG_Idle:
656 run_petitmodbus = true;
657 break;
658 case eCFG_Cache:
659 run_petitmodbus = true;
660 mbWDTen = false;
661 // the magic in processing actually happens in the modbus rx function
662 break;
663 case eCFG_Commit:
664 run_petitmodbus = true;
665 // functionality implemented in the transitions, believe it or not
666 break;
667 case eCFG_Erase:
668 run_petitmodbus = false;
669 FLASH_PageErase(C_FLASH_CONF);
670 break;
671 case eCFG_Write:
672 run_petitmodbus = false;
673 cfg_write();
674 break;
675 }
