55 TMOD = (TMOD & ~TMOD_T1M__FMASK) | TMOD_T1M__MODE3;
56
57 // decode T1 clock source

64 {
65 // system clock
66 CKCON0 = (CKCON0 & ~CKCON0_T1M__BMASK) | CKCON0_T1M__SYSCLK;
67 }

70 TMOD = (TMOD & ~TMOD_T1M__FMASK) | TMOD_T1M__MODE2;
71
72 // set timer 1 current value and reload value
73 TH1 = (mmw_ct[baud - 1][eMMW_CT_TH1] << TH1_TH1__SHIFT);
